 39      * When there are multiple arrays and they have similar index structures but it's not certain if one has the key and the others,

 41      * 
 42      * @remark          This is mainly used by the field array to insert user-defined key values.
 43      * @return          string|array            If the key does not exist in the passed array, it will return the default. If the subject value is not an array, it will return the subject value itself.

 48     public static function getCorrespondingArrayValue( $vSubject, $sKey, $sDefault='', $bBlankToDefault=false ) {   
 49                 
 50         // If $vSubject is null,
 51         if ( ! isset( $vSubject ) ) return $sDefault;   
 52             
 53         // If the $bBlankToDefault flag is set and the subject value is a blank string, return the default value.
 54         if ( $bBlankToDefault && $vSubject == '' ) return $sDefault;
 55             
 56         // If $vSubject is not an array, 
 57         if ( ! is_array( $vSubject ) ) return ( string ) $vSubject; // consider it as string.
 58         
 59         // Consider $vSubject as array.
 60         if ( isset( $vSubject[ $sKey ] ) ) return $vSubject[ $sKey ];

102     /**
103      * Merges two multi-dimensional arrays recursively.
104      * 
105      * The first parameter array takes its precedence. This is useful to merge default option values. 
106      * An alternative to <em>array_replace_recursive()</em>; it is not supported PHP 5.2.x or below.
107      * 
108      * @since           2.0.0
109      * @since           2.1.5               Changed the scope to static. 
110      * @access          public
111      * @remark          null values will be overwritten.    
112      * @param           array           the array that overrides the same keys.
113      * @param           array           the array that is going to be overridden.
114      * @return          array           the united array.
115      */ 
116     public static function uniteArraysRecursive( $aPrecedence, $aDefault ) {
117                 
118         if ( is_null( $aPrecedence ) ) $aPrecedence = array();
119         
120         if ( ! is_array( $aDefault ) || ! is_array( $aPrecedence ) ) return $aPrecedence;
121             
122         foreach( $aDefault as $sKey => $v ) {
123             
124             // If the precedence does not have the key, assign the default's value.
125             if ( ! array_key_exists( $sKey, $aPrecedence ) || is_null( $aPrecedence[ $sKey ] ) )
126                 $aPrecedence[ $sKey ] = $v;
127             else {
128                 
129                 // if the both are arrays, do the recursive process.
130                 if ( is_array( $aPrecedence[ $sKey ] ) && is_array( $v ) ) 
131                     $aPrecedence[ $sKey ] = self::uniteArraysRecursive( $aPrecedence[ $sKey ], $v );            
132             
133             }
134         }
135         return $aPrecedence;        
136     }

171     /**
172      * Calculates the relative path from the given path.
173      * 
174      * This function is used to generate a template path.
175      * 
176      * @since           2.1.5
177      * @author          Gordon
178      * @author          Michael Uno,            Modified variable names and spacing.
179      * @see             http://stackoverflow.com/questions/2637945/getting-relative-path-from-absolute-path-in-php/2638272#2638272
180      */
181     static public function getRelativePath( $from, $to ) {
182         
183         // some compatibility fixes for Windows paths
184         $from = is_dir( $from ) ? rtrim( $from, '\/') . '/' : $from;
185         $to   = is_dir( $to )   ? rtrim( $to, '\/') . '/'   : $to;
186         $from = str_replace( '\\', '/', $from );
187         $to   = str_replace( '\\', '/', $to );
188 
189         $from     = explode( '/', $from );
190         $to       = explode( '/', $to );
191         $relPath  = $to;
192 
193         foreach( $from as $depth => $dir ) {
194             // find first non-matching dir
195             if( $dir === $to[ $depth ] ) {
196                 // ignore this directory
197                 array_shift( $relPath );
198             } else {
199                 // get number of remaining dirs to $from
200                 $remaining = count( $from ) - $depth;
201                 if( $remaining > 1 ) {
202                     // add traversals up to first matching dir
203                     $padLength = ( count( $relPath ) + $remaining - 1 ) * -1;
204                     $relPath = array_pad( $relPath, $padLength, '..' );
205                     break;
206                 } else {
207                     $relPath[ 0 ] = './' . $relPath[ 0 ];
208                 }
209             }
210         }
211         return implode( '/', $relPath );
212         
213     }

244     /**
245      * Generates the string of attributes to be embedded in an HTML tag from an associative array.
246      * 
247      * For example, 
248      *  array( 'id' => 'my_id', 'name' => 'my_name', 'style' => 'background-color:#fff' )
249      * becomes
250      *  id="my_id" name="my_name" style="background-color:#fff"
251      * 
252      * This is mostly used by the method to output input fields.
253      * @since           3.0.0
254      */
255     static public function generateAttributes( array $aAttributes ) {
256         
257         $aOutput = array();
258         foreach( $aAttributes as $sAttribute => $sProperty ) {
259             if ( empty( $sProperty ) && $sProperty !== 0  ) continue;   // drop non-value elements.
260             if ( is_array( $sProperty ) || is_object( $sProperty ) ) continue;  // must be resolved as a string.
261             $aOutput[] = "{$sAttribute}='{$sProperty}'";
262         }
263         return implode( ' ', $aOutput );
264         
265     }
