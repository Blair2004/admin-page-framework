252         // If $this->aOptions is called for the first time, retrieve the option data from the database and assign to the property.

255         if ( $sName == 'aOptions' ) {

276         
277         $sPageSlug = $sPageSlug ? $sPageSlug : ( isset( $_GET['page'] ) ? $_GET['page'] : '' );

301     /**
302      * Retrieves the framework's saved options from the options table of the database.
303      * @since           2.0.0
304      */
305     public function getOptions() {
306         
307         $vOptions = get_option( $this->sOptionKey );
308         if ( empty( $vOptions ) ) return array();       // casting array causes a 0 key element when the value is empty. So this way it can be avoided
309         if ( is_array( $vOptions ) ) return $vOptions;  // if it's array, no problem.       
310         return ( array ) $vOptions; // finally cast array.
311         
312     }
